**Step 1: Fetch and validate 5 JSON-LD blocks**

```bash

import sys, re, json

html = sys.stdin.read()

# Extract JSON-LD blocks
blocks = re.findall(r'<script type=\"application/ld\+json\">(.*?)</script>', html, re.DOTALL)
print(f'JSON-LD blocks found: {len(blocks)} (expected: 5)')
assert len(blocks) == 5, f'FAIL: Expected 5 JSON-LD blocks, found {len(blocks)}'

expected_types = {'Organization', 'WebSite', 'WebPage', 'Service', 'FAQPage'}
found_types = set()

for i, block in enumerate(blocks):
    data = json.loads(block)
    schema_type = data.get('@type', 'Unknown')
    found_types.add(schema_type)
    print(f'  Block {i+1}: {schema_type} - VALID JSON')

    # Type-specific checks
    if schema_type == 'Organization':
        assert data.get('name') == 'AutoDokas', f'Organization name mismatch: {data.get(\"name\")}'
        print(f'    name: {data[\"name\"]} - OK')

    elif schema_type == 'Service':
        has_broker = 'broker' in json.dumps(data).lower()
        print(f'    Broker property present: {has_broker}')
        assert has_broker, 'Service missing Broker property'

    elif schema_type == 'FAQPage':
        main_entity = data.get('mainEntity', [])
        q_count = len(main_entity)
        print(f'    FAQ questions: {q_count} (expected: 7)')
        assert q_count == 7, f'Expected 7 FAQ questions, found {q_count}'

missing = expected_types - found_types
assert not missing, f'Missing schema types: {missing}'
print(f'\nAll 5 schema types present: {sorted(found_types)}')
print('PASS: All JSON-LD schemas valid')
"
```

**Step 2: Validate meta tags**

```bash
curl -s https://autodokas.lt/ | python3 -c "
import sys, re

html = sys.stdin.read()
checks = []

# Meta description
desc_match = re.search(r'<meta\s+name=\"description\"\s+content=\"([^\"]+)\"', html)
if desc_match:
    desc = desc_match.group(1)
    checks.append(('Meta description present', True, desc[:60] + '...'))
    checks.append(('Meta description length <= 160', len(desc) <= 160, f'{len(desc)} chars'))
else:
    checks.append(('Meta description present', False, 'NOT FOUND'))

# Canonical URL
canon_match = re.search(r'<link\s+rel=\"canonical\"\s+href=\"([^\"]+)\"', html)
if canon_match:
    canon = canon_match.group(1)
    checks.append(('Canonical URL', canon == 'https://autodokas.lt/', canon))
else:
    checks.append(('Canonical URL', False, 'NOT FOUND'))

# Open Graph tags
og_tags = {'og:title': None, 'og:description': None, 'og:url': None, 'og:type': 'website', 'og:site_name': 'AutoDokas'}
for tag, expected in og_tags.items():
    match = re.search(rf'<meta\s+property=\"{tag}\"\s+content=\"([^\"]+)\"', html)
    if match:
        val = match.group(1)
        if expected:
            checks.append((f'{tag}', val == expected, val))
        else:
            checks.append((f'{tag} present', True, val[:50]))
    else:
        checks.append((f'{tag}', False, 'NOT FOUND'))

# HTML lang attribute
lang_match = re.search(r'<html[^>]+lang=\"([^\"]+)\"', html)
if lang_match:
    lang = lang_match.group(1)
    checks.append(('HTML lang attribute', lang == 'lt', lang))
else:
    checks.append(('HTML lang attribute', False, 'NOT FOUND'))

# Print results
all_pass = True
for name, passed, detail in checks:
    status = 'PASS' if passed else 'FAIL'
    if not passed: all_pass = False
    print(f'{status}  {name}: {detail}')

assert all_pass, 'Some meta tag checks failed'
print('\nPASS: All meta tags correct')
"
```

for tag, text in headings:

        print(f'  {tag}: {clean[:60]}')

# Check no h4, h5, h6 content headings (h4-h6 may appear in framework chrome, check main content)

    print(f'\nWARNING: Found {len(h4_h5_h6)} h4/h5/h6 headings -- verify these are not content headings')
